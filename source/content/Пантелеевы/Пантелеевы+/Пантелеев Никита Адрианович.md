# Источники 

```dataviewjs
const pages = dv.pages('"Источники"')
  .where(p => p.file.ext === "md" && (() => {
    const персоны = p["Персоны"];
    if (!персоны) return false;
    return Array.isArray(персоны) 
      ? персоны.some(link => link.path === dv.current().file.path)
      : персоны.path === dv.current().file.path;
  })())
  .sort(p => p["Год"], 'asc');

for (let page of pages) {
  // Определяем значение для заголовка: "Дата события" если заполнено, иначе "Год"
  const год = page["Год"] || '';
  const датаСобытия = page["Дата события"] || '';
  const заголовок = датаСобытия || год;
  
  dv.header(3, `${заголовок} - ` + page.file.link);
  
  // Загружаем контент БЕЗ frontmatter и свойств
  const content = await dv.io.load(page.file.path);
  const cleanContent = content.replace(/^---[\s\S]*?---\n?/i, '').trim();
  
  dv.paragraph(cleanContent);
  dv.paragraph("---");
}
```
